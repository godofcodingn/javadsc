    FibLength = sc.nextInt();
    int[] num = new int[FibLength];

    }
    }
    }

    • 12.EXCEPTION HANDLING

  • import java.util.Scanner;
    public class man {
    public static double man(double a, double b) throws ArithmeticException {
    if (b == 0) {
    throw new ArithmeticException(" by zero is not allowed");
    }
    return a / b;
    }
    public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    System.out.print("Enter the numerator (a): ");
    double a = scanner.nextDouble();
    System.out.print("Enter the denominator (b): ");
    double b = scanner.nextDouble();
    try {
    double divisionResult = man(a, b);
    System.out.println("The result of " + a + " divided by " + b + " is: " + divisionResult);
    } catch (ArithmeticException e) {
    System.out.println("Error: " + e.getMessage());
    } finally {
    scanner.close();
    }
    }
    }
